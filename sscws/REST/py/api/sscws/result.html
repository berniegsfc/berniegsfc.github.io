287    def get_data_result(
288            data_result_element: ET
289        ) -> Dict:
290        """
291        Produces a Dict representation of a DataResult from the given
292        xml representation of a DataResult.
293
294        Parameters
295        ----------
296        data_result_element
297            ElementTree representation of a DataResult from
298            <https://sscweb.gsfc.nasa.gov/WebServices/REST/SSC.xsd>.
299
300        Returns
301        -------
302        Dict
303            Dict representation of the given ElementTree DataResult
304            as described in
305            <https://sscweb.gsfc.nasa.gov/WebServices/REST/SSC.xsd>.
306
307        Raises
308        ------
309        ValueError
310            If the given xml is not a valid XML representation of a
311            DataResult.
312        """
313
314        result = Result.get_status(data_result_element)
315
316        data_elements = data_result_element.findall('ssc:Data', namespaces=NS)
317        if data_elements is not None:
318            data_len = len(data_elements)
319        else:
320            data_len = 0
321        result['Data'] = np.empty(data_len, dtype=object)
322
323        for data_i, data_element in enumerate(data_elements):
324
325            result['Data'][data_i] = {
326                'Id': data_element.find('ssc:Id', namespaces=NS).text,
327            }
328            coords_elements = data_element.findall('ssc:Coordinates',
329                                                   namespaces=NS)
330
331            if coords_elements is not None:
332                coords_len = len(coords_elements)
333            else:
334                coords_len = 0
335            result['Data'][data_i]['Coordinates'] =\
336                np.empty(coords_len, dtype=object)
337
338            for coords_i, coords_element in enumerate(coords_elements):
339
340                coord_system = coords_element.find(\
341                            'ssc:CoordinateSystem', namespaces=NS).text
342
343                result['Data'][data_i]['Coordinates'][coords_i] = {
344                    'CoordinateSystem': CoordinateSystem(coord_system)
345                }
346
347                for name in ['X', 'Y', 'Z', 'Latitude', 'Longitude',
348                             'LocalTime']:
349                    result['Data'][data_i]['Coordinates'][coords_i][name] = \
350                        Result._get_data(coords_element, name, float)
351
352            result['Data'][data_i]['Time'] = \
353                Result._get_data(data_element, 'Time', datetime)
354
355
356            b_trace_data_elements = data_element.findall(\
357                   'ssc:BTraceData', namespaces=NS)
358            if b_trace_data_elements is not None:
359                b_trace_data_len = len(b_trace_data_elements)
360            else:
361                b_trace_data_len = 0
362
363            result['Data'][data_i]['BTraceData'] = \
364                np.empty(b_trace_data_len, dtype=object)
365
366            for b_trace_data_i, b_trace_data in \
367                enumerate(b_trace_data_elements):
368
369                result['Data'][data_i]['BTraceData'][b_trace_data_i] = {
370                    'CoordinateSystem': CoordinateSystem(\
371                        b_trace_data.find(\
372                            'ssc:CoordinateSystem', namespaces=NS).text),
373                    'Hemisphere': Hemisphere(b_trace_data.find(\
374                            'ssc:Hemisphere', namespaces=NS).text),
375                }
376                for name in ['Latitude', 'Longitude', 'ArcLength']:
377                    result['Data'][data_i]['BTraceData'][b_trace_data_i][name] = \
378                        Result._get_data(b_trace_data, name, float)
379
380
381            for name in ['RadialLength', 'MagneticStrength',
382                         'NeutralSheetDistance', 'BowShockDistance',
383                         'MagnetoPauseDistance', 'DipoleLValue',
384                         'DipoleInvariantLatitude']:
385                result['Data'][data_i][name] = \
386                    Result._get_data(data_element, name, float)
387
388
389            result['Data'][data_i]['SpacecraftRegion'] = \
390                Result._get_data(data_element, 'SpacecraftRegion',
391                                 SpaceRegion)
392
393            for name in ['BGseX', 'BGseY', 'BGseZ']:
394                result['Data'][data_i][name] = \
395                    Result._get_data(data_element, name, float)
396
397            for name in ['RadialTracedFootpointRegions',
398                         'NorthBTracedFootpointRegions',
399                         'SouthBTracedFootpointRegions']:
400                result['Data'][data_i][name] = \
401                    Result._get_data(data_element, name,
402                                     FootpointRegion)
403
404        return result

409    @staticmethod
410    def _get_data(
411            data_element: ET,
412            name: str,
413            value_type: Callable[[str], Any],
414        ) -> np.ndarray:
415        """
416        Produces an np.ndarray(m, dtype=value_type) representation of the
417        values of specified element in the given data_element ET.
418
419        Parameters
420        ----------
421        data_element
422            ElementTree representation of a DataResult from
423            <https://sscweb.gsfc.nasa.gov/WebServices/REST/SSC.xsd>.
424        name
425            Name of element in data_element to get values of.
426        value_type
427            Type of values to get.
428
429        Returns
430        -------
431        np.ndarray
432            np.ndarray representation of the specified values from
433            data_element.
434        """
435        data_elements = data_element.findall('ssc:' + name, namespaces=NS)
436        if data_elements is not None:
437            size = len(data_elements)
438        else:
439            size = 0
440        values = np.empty(size, dtype=value_type)
441        for index, value in enumerate(data_elements):
442            if value_type is datetime:
443                values[index] = dateutil.parser.parse(value.text)
444            else:
445                values[index] = value_type(value.text)
446
447        return values
