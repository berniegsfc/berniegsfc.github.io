 233    def get_observatories(
 234            self,
 235            obs_ids: Optional[List[str]] = None
 236        ) -> Dict:
 237        """
 238        Gets a description of the available SSC observatories.
 239
 240        Parameters
 241        ----------
 242        obs_ids
 243            Optional list of observatory identifiers to get.  If none
 244            are given, all are returned.
 245
 246        Returns
 247        -------
 248        Dict
 249            Dictionary whose structure mirrors ObservatoryResponse from
 250            <https://sscweb.gsfc.nasa.gov/WebServices/REST/SSC.xsd>
 251            with the addition of the following key/values:<br>
 252            - HttpStatus: with the value of the HTTP status code.
 253              Successful == 200.<br>
 254            When HttpStatus == 200:<br>
 255            - Observatories: containing a List of Dictionaries describing
 256              each observatory (keys: Id, Name, Resolution, StartTime,
 257              EndTime, ResourceId).<br>
 258            When HttpStatus != 200:<br>
 259            - HttpText: containing a string representation of the HTTP
 260              entity body.<br>
 261            When HttpText is a standard SSC WS error entity body the
 262            following key/values (convenience to avoid parsing
 263            HttpStatus):<br>
 264            - ErrorMessage: value from HttpText.<br>
 265            - ErrorDescription: value from HttpText.<br>
 266        """
 267        url = self._endpoint + 'observatories'
 268        params = []
 269
 270        if obs_ids is None:
 271            obs_ids = []
 272        for obs_id in obs_ids:
 273            params.append(('id', obs_id))
 274
 275        headers = None
 276        if self._cache['Observatories']['ETag'] is not None:
 277            headers = {
 278                'If-None-Match': self._cache['Observatories']['ETag']
 279            }
 280
 281        response = self._session.get(url, timeout=self._timeout,
 282                                     headers=headers, params=params)
 283
 284        #self.logger.debug(f'url = {response.url}, '\
 285        #    'status = {response.status_code}')
 286
 287        if response.status_code == 304:
 288            return self._cache['Observatories']['Value']
 289
 290        status = self.__get_status(response)
 291        if response.status_code != 200:
 292            self.logger.warning('%s failed with status %d', response.url,
 293                                response.status_code)
 294            return status
 295
 296        observatory_response = ET.fromstring(response.text)
 297
 298        result = {
 299            'Observatory': []
 300        }
 301
 302        for observatory in observatory_response.findall('ssc:Observatory',
 303                                                        namespaces=NS):
 304            result['Observatory'].append({
 305                'Id': observatory.find('ssc:Id', namespaces=NS).text,
 306                'Name': observatory.find('ssc:Name', namespaces=NS).text,
 307                'Resolution': int(observatory.find('ssc:Resolution',
 308                                                   namespaces=NS).text),
 309                'StartTime': dateutil.parser.parse(observatory.find(\
 310                    'ssc:StartTime', namespaces=NS).text),
 311                'EndTime': dateutil.parser.parse(observatory.find(\
 312                    'ssc:EndTime', namespaces=NS).text),
 313                'ResourceId': observatory.find('ssc:ResourceId',
 314                                               namespaces=NS).text
 315            })
 316
 317        result.update(status)
 318
 319        if 'ETag' in response.headers:
 320            etag = response.headers['ETag']
 321            # workaround old apache bugs that are still causing problems
 322            etag = etag.replace('-gzip', '')
 323            self._cache['Observatories']['ETag'] = etag
 324            self._cache['Observatories']['Value'] = result
 325        else:
 326            self._cache['Observatories']['ETag'] = None
 327            self._cache['Observatories']['Value'] = None
 328
 329        return result

 332    def get_ground_stations(
 333            self
 334        ) -> Dict:
 335        """
 336        Gets a description of the available SSC ground stations.
 337
 338        Returns
 339        -------
 340        Dict
 341            Dictionary whose structure mirrors GroundStationResponse from
 342            <https://sscweb.gsfc.nasa.gov/WebServices/REST/SSC.xsd>
 343            with the addition of the following key/values:<br>
 344            - HttpStatus: with the value of the HTTP status code.
 345              Successful == 200.<br>
 346            When HttpStatus != 200:<br>
 347            - HttpText: containing a string representation of the HTTP
 348              entity body.<br>
 349            When HttpText is a standard SSC WS error entity body the
 350            following key/values (convenience to avoid parsing
 351            HttpStatus):<br>
 352            - ErrorMessage: value from HttpText.<br>
 353            - ErrorDescription: value from HttpText.<br>
 354        """
 355        url = self._endpoint + 'groundStations'
 356
 357        self.logger.debug('request url = %s', url)
 358
 359        headers = None
 360        if self._cache['GroundStations']['Last-Modified'] is not None:
 361            headers = {
 362                'If-Modified-Since': self._cache['GroundStations']['Last-Modified']
 363            }
 364
 365        response = self._session.get(url, timeout=self._timeout,
 366                                     headers=headers)
 367
 368        if response.status_code == 304:
 369            return self._cache['GroundStations']['Value']
 370
 371        status = self.__get_status(response)
 372        if response.status_code != 200:
 373            return status
 374
 375        ground_station_response = ET.fromstring(response.text)
 376
 377        result = {
 378            'GroundStation': []
 379        }
 380
 381        for ground_station in ground_station_response.findall(\
 382                'ssc:GroundStation', namespaces=NS):
 383
 384            location = ground_station.find('ssc:Location', namespaces=NS)
 385            latitude = float(location.find('ssc:Latitude', namespaces=NS).text)
 386            longitude = float(location.find('ssc:Longitude',
 387                                            namespaces=NS).text)
 388
 389            result['GroundStation'].append({
 390                'Id': ground_station.find('ssc:Id', namespaces=NS).text,
 391                'Name': ground_station.find('ssc:Name', namespaces=NS).text,
 392                'Location': {
 393                    'Latitude': latitude,
 394                    'Longitude': longitude
 395                }
 396            })
 397
 398        result.update(status)
 399
 400        if 'Last-Modified' in response.headers:
 401            self._cache['GroundStations']['Last-Modified'] = response.headers['Last-Modified']
 402            self._cache['GroundStations']['Value'] = result
 403        else:
 404            self._cache['GroundStations']['Last-Modified'] = None
 405            self._cache['GroundStations']['Value'] = None
 406
 407        return result

 499            self,
 500            param1: Union[List[str], DataRequest],
 501            time_range: Union[List[str], TimeInterval] = None,
 502            coords: List[CoordinateSystem] = None
 503        ) -> Dict:
 504        """
 505        Gets the specified locations using CDF instead of XML.  This
 506        method is faster, supports larger requests, and the server

 508        `SscWs.get_locations` method.  Complex requests (requesting
 509        magnetic field model values) require a single DataRequest
 510        parameter.  Simple requests (for only x, y, z, lat, lon,
 511        local_time) require at least the first two parameters.
 512
 513        Parameters
 514        ----------
 515        param1
 516            A locations DataRequest or a list of observatory identifier
 517            (returned by get_observatories).
 518        time_range
 519            A TimeInterval or two element array of ISO 8601 string
 520            values of the start and stop time of requested data.  The
 521            datetime values should have a UTC timezone.  If the values
 522            have no timezone, it will be set to UTC.  A datetime with
 523            a non-UTC timezone, will have its value adjusted to UTC and
 524            the returned data may not have the expected range.
 525        coords
 526            Array of CoordinateSystem values that location information
 527            is to be in.  If None, default is CoordinateSystem.GSE.
 528
 529        Returns
 530        -------
 531        Dict
 532            Dictionary whose structure mirrors Result from
 533            <https://sscweb.gsfc.nasa.gov/WebServices/REST/SSC.xsd>
 534            with the addition of the following key/values:<br>
 535            - HttpStatus: with the value of the HTTP status code.
 536              Successful == 200.<br>
 537            When HttpStatus != 200:<br>
 538            - HttpText: containing a string representation of the HTTP
 539              entity body.<br>
 540            When HttpText is a standard SSC WS error entity body the
 541            following key/values (convenience to avoid parsing
 542            HttpStatus):<br>
 543            - ErrorMessage: value from HttpText.<br>
 544            - ErrorDescription: value from HttpText.<br>
 545
 546        Raises
 547        ------
 548        ValueError
 549            If param1 is not a DataRequest and time_range is missing or
 550            time_range does not contain valid values.
 551        ModuleNotFoundExcepttion
 552            If cdflib is not installed.
 553
 554        Warnings
 555        --------
 556        This method is experimental and may be eliminated or changed
 557        significantly in future releases.  Code expecting a stable API
 558        should use `SscWs.get_locations` instead.  The results returned 
 559        are compatible with `SscWs.get_locations` except that the numpy 
 560        array of datetime.datetime values are returned as a numpy array 
 561        of numpy.datetime64 values.  This method requires the 
 562        cdflib module to be installed.
 563
 564        See Also
 565        --------
 566        SscWs.get_locations : Gets the specified locations.

 572        if isinstance(param1, DataRequest):
 573            request = param1
 574        else:
 575            request = SscWs.__create_locations_request(param1, time_range,
 576                                                       coords)
 577        if request.format_options is None:
 578            request.format_options = CdfFormatOptions()
 579
 580        request.format_options.cdf = True
 581
 582        result = self.__get_locations(request)
 583
 584        if result['HttpStatus'] == 200:
 585            #print('result = ', result)
 586            result = self.get_locations_from_file(result)
 587
 588        return result

 591    def download(
 592            self,
 593            url: str
 594        ) -> str:
 595        """
 596        Downloads the file specified by the given URL to a temporary
 597        file without reading all of it into memory.  This method
 598        utilizes the connection pool and persistent HTTP connection
 599        to the SscWs server.
 600
 601        Parameters
 602        ----------
 603        url
 604            URL of file to download.
 605        Returns
 606        -------
 607        str
 608            name of tempory file or None if there was an error.
 609        """
 610        suffix = os.path.splitext(urlparse(url).path)[1]
 611
 612        file_descriptor, tmp_filename = mkstemp(suffix=suffix)
 613
 614        with self._session.get(url, stream=True,
 615                               timeout=self._timeout) as response:
 616
 617            with open(tmp_filename, 'wb') as file:
 618                for chunk in response.iter_content(chunk_size=8192):
 619                    if chunk:  # filter out keep-alive new chunks
 620                        file.write(chunk)
 621            os.close(file_descriptor)
 622
 623        return tmp_filename

 626    def get_locations_from_file(
 627            self,
 628            results: Result
 629        ) -> Dict:
 630        """
 631        Gets the given file(s) from the server and returns the contents
 632        in a dictionary.
 633
 634        Parameters
 635        ----------
 636        results
 637            results to get locations from.
 638
 639        Returns
 640        -------
 641        Dict
 642            Dictionary whose structure mirrors Result from
 643            <https://sscweb.gsfc.nasa.gov/WebServices/REST/SSC.xsd>
 644            with the addition of the following key/values:<br>
 645            - HttpStatus: with the value of the HTTP status code.
 646              Successful == 200.<br>
 647            When HttpStatus != 200:<br>
 648            - HttpText: containing a string representation of the HTTP
 649              entity body.<br>
 650            When HttpText is a standard SSC WS error entity body the
 651            following key/values (convenience to avoid parsing
 652            HttpStatus):<br>
 653            - ErrorMessage: value from HttpText.<br>
 654            - ErrorDescription: value from HttpText.<br>
 655        """
 656        locations_result = {}
 657        locations_result['HttpStatus'] = results['HttpStatus']
 658        locations_result['StatusCode'] = results['StatusCode']
 659        locations_result['StatusSubCode'] = results['StatusSubCode']
 660        locations_result['Data'] = np.empty(len(results['Files']), dtype=object)
 661        #print('results = ', results)
 662        for index in range(len(results['Files'])):
 663            tmp_cdf_file = 'unset'
 664            try:
 665                result_url = results['Files'][index]['Name']
 666                tmp_cdf_file = self.download(result_url)
 667                #print('tmp_cdf_file', tmp_cdf_file)
 668                cdf = Cdf()
 669                cdf.open(tmp_cdf_file)
 670                locations_result['Data'][index] = cdf.get_satellite_data()
 671                #cdf.close() ???
 672                os.remove(tmp_cdf_file)
 673                #print('tmp_cdf_file', tmp_cdf_file, 'retained')
 674            except:
 675                self.logger.error('Exception from read_data(%s): %s',
 676                                  tmp_cdf_file, sys.exc_info()[0])
 677                self.logger.error('CDF file has been retained.')
 678                raise
 679        return locations_result

 592    def download(
 593            self,
 594            url: str
 595        ) -> str:
 596        """
 597        Downloads the file specified by the given URL to a temporary
 598        file without reading all of it into memory.  This method
 599        utilizes the connection pool and persistent HTTP connection
 600        to the SscWs server.
 601
 602        Parameters
 603        ----------
 604        url
 605            URL of file to download.
 606        Returns
 607        -------
 608        str
 609            name of tempory file or None if there was an error.
 610        """
 611        suffix = os.path.splitext(urlparse(url).path)[1]
 612
 613        file_descriptor, tmp_filename = mkstemp(suffix=suffix)
 614
 615        with self._session.get(url, stream=True,
 616                               timeout=self._timeout) as response:
 617
 618            with open(tmp_filename, 'wb') as file:
 619                for chunk in response.iter_content(chunk_size=8192):
 620                    if chunk:  # filter out keep-alive new chunks
 621                        file.write(chunk)
 622            os.close(file_descriptor)
 623
 624        return tmp_filename

 726            obs_ids: List[str],
 727            time_range: Union[List[str], TimeInterval] = None,
 728            coords: List[CoordinateSystem] = None
 729        ) -> DataRequest:
 730        """
 731        Creates a "simple" (only x, y, z, lat, lon, local_time in GSE)
 732        locations DataRequest for the given values.
 733        More complicated requests should be made with DataRequest
 734        directly.
 735
 736        Parameters
 737        ----------

 739            A list of observatory identifier (returned by

 752        ------
 753        ValueError
 754            If time_range is missing or time_range does not contain
 755            valid values.
 756        """
 757
 758        sats = []
 759        for sat in obs_ids:
 760            sats.append(SatelliteSpecification(sat, 1))
 761
 762        if time_range is None:
 763            raise ValueError('time_range value is required when ' +
 764                             '1st is not a DataRequest')
 765
 766        if isinstance(time_range, list):
 767            time_interval = TimeInterval(time_range[0], time_range[1])
 768        else:
 769            time_interval = time_range
 770

 772            coords = [CoordinateSystem.GSE]

 775        for coord in coords:
 776            coord_options.append(
 777                CoordinateOptions(coord, CoordinateComponent.X))

 779                CoordinateOptions(coord, CoordinateComponent.Y))
 780            coord_options.append(
 781                CoordinateOptions(coord, CoordinateComponent.Z))
 782            coord_options.append(
 783                CoordinateOptions(coord, CoordinateComponent.LAT))
 784            coord_options.append(
 785                CoordinateOptions(coord, CoordinateComponent.LON))
 786            coord_options.append(
 787                CoordinateOptions(coord, CoordinateComponent.LOCAL_TIME))
 788
 789        return DataRequest(None, time_interval, sats, None,
 790                           OutputOptions(coord_options), None, None)
